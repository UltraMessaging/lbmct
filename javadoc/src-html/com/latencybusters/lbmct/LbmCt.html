032/**
033 * An instance of the Connected Topics for Ultra Messaging (UM).
034 * <p>
035 * A Connected Topics instance is associated with a UM context.
036 * A given UM context must not have more than one Connected Topics instance associated with it.
037 * When an {@code LbmCt} is created, its full initialization is deferred until its {@link #start} method is called.
038 * <p>
039 * A Connected Topics instance has an independent thread associated with it.
040 * Since it is an active object, it must be explicitly deleted when it is no longer needed, using the {@link #close} method.
041 */

084  // THREAD: user
085  @SuppressWarnings("WeakerAccess")  // public API.
086  public void start(LBMContext inCtx, LbmCtConfig inConfig, ByteBuffer inMetadata) throws Exception {
087    ctx = inCtx;
088    if (inConfig == null) {
089      throw new LBMException("config must be non-null");
090    }
091    config = inConfig;
092
093    // Generate a random 32-bit ID from system entropy.  Used to make a unique ID for this LbmCt instance.
094    SecureRandom secRand = new SecureRandom();
095    byte[] randomSeed = secRand.generateSeed(4);
096    // Make the 4 byte into an int.
097    ByteBuffer randomByteBuf = ByteBuffer.wrap(randomSeed);
098    ctId = randomByteBuf.getInt();
099    if (ctId == 0) {
100      throw new LBMException("generateSeed failed");
101    }
102
103    if (inMetadata == null) {  // Empty metadata.
104      metadata = ByteBuffer.allocate(1);
105      metadata.clear();
106      metadata.flip();
107    } else {
108      if (inMetadata.position() != 0) {
109        throw new LBMException("Metadata byte buffer position should be 0, missing flip?");
110      }
111
112      // Make a deep copy of the user's metadata.  Don't assume position is zero.  Data
113      // is between position and limit, which would normally be 0..limit.
114      int oldPosition = inMetadata.position();
115      int metadataLen = inMetadata.remaining();
116      metadata = ByteBuffer.allocate(metadataLen);
117      // Read the data out of the passed-in metadata.
118      inMetadata.get(metadata.array(), 0, metadataLen);
119      metadata.position(metadataLen);
120      inMetadata.position(oldPosition);  // Restore input metadata byte buffer (don't "consume" it).
121      metadata.flip();
122    }
123
124    localUimAddr();
125
126    srcExInfo = new LBMSourceSendExInfo();
127    LBMMessageProperties msgProps = new LBMMessageProperties();
128    msgProps.set(HANDSHAKE_TOPIC_STR, 1);
129    srcExInfo.setMessageProperties(msgProps);
130
131    handshakeParser = new LbmCtHandshakeParser(this.metadata.remaining());
132
133    ctSrcMap = new HashMap<>();
134    srcConnMap = new HashMap<>();
135
136    ctSrcSet = new HashSet<>();
137    ctRcvSet = new HashSet<>();
138
139    // Create and start ct control thread.
140    ctrlr = new LbmCtCtrlr(this);
141    ctrlr.start();
142
143    // Create UM receiver for source-side reception of handshake messages.
144    srcHandshakeRcvCreate();
145  }

149  // THREAD: user
150  @SuppressWarnings("WeakerAccess")  // public API.
151  public void close() throws Exception {
152    if (! ctSrcSet.isEmpty()) {
153      throw new LBMException("Must delete sources and receivers");
154    }
155    if (! ctRcvSet.isEmpty()) {
156      throw new LBMException("Must delete receives and sources");
157    }
158
159    // CT sources and receivers are gone.  Tear down everything else.
160    umHandshakeRcv.close();  // Delete the UM receiver.
161    ctrlr.close();
162    ctrlr.join();
163
164    // Remove references to everything.
165    ctx = null;  // Initialized in constructor.
166    config = null;
167    metadata = null;
168    srcExInfo = null;
169    ctrlr = null;
170    umHandshakeRcv = null;
171    handshakeParser = null;
172    ctSrcMap = null;
173    srcConnMap = null;
174    ctSrcSet = null;
175    ctRcvSet = null;
176  }

179  // Send a test message into the ct control thread and get the response back.
180  // THREAD: user
181  void ctTest(String logText, int testErr) throws Exception {
182    LbmCtCtrlrCmd cmd = ctrlr.cmdGet();
183    cmd.setTest(this, logText, testErr);
184    ctrlr.submitWait(cmd);  // This "calls" cmdTest below.
185
186    Exception e = cmd.getE();
187    ctrlr.cmdFree(cmd);  // Return command object to free pool.
188    if (e != null) {
189      throw (new Exception(e));
190    }
191  }

193  // THREAD: ctrlr
194  boolean cmdTest(LbmCtCtrlrCmd cmd) throws Exception {
195    // For testing, throw an exception if the string is null.  This lets the test program verify proper exception flow.
196    int testErr = cmd.getTestErr();
197    String testStr = cmd.getTestStr();
198    if (testErr != 0) {
199      throw new LBMException("cmdTest: " + testStr);
200    }
201    LBMPubLog.pubLog(LBM.LOG_INFO, "cmdTest: " + cmd.getTestStr() + "\n");
202    return true;
203  }

239  // Determine our context's UIM destination address by looking at the context config attrlbutes.
240  // The destination is loaded into binary variables "localDomainId", "localRequestPort", and "localIpAddr".
241  // THREAD: user
242  private void localUimAddr() throws Exception {
243    String is_bound = ctx.getAttributeValue("request_tcp_bind_request_port");
244    if (!is_bound.contentEquals("1")) {
245      throw new LBMException("context request_tcp_bind_request_port must not be 0");
246    }
247
248    localRequestPort = Integer.parseInt(ctx.getAttributeValue("request_tcp_port"));
249
250    localDomainId = config.getDomainId();
251
252    String localIpAddrStr = ctx.getAttributeValue("request_tcp_interface");
253    if (localIpAddrStr.contentEquals("0.0.0.0")) {
254      // No unicast interface specified, so any "should" work.  Use the
255      // multicast resolver interface, since that defaults to *something*.
256      localIpAddrStr = ctx.getAttributeValue("resolver_multicast_interface");
257    }
258
259    // Have a string form of ip address.  Change to big endian integer (first octect as MSB).
260    // Write each octet as a byte to a byte buffer, then read it back as an int.
261    Scanner ipScanner = new Scanner(localIpAddrStr).useDelimiter("\\.");
262    ByteBuffer ipByteBuf = ByteBuffer.allocate(4);
263    for (int i = 0; i < 4; i++) {
264      ipByteBuf.put((byte)ipScanner.nextInt());
265    }
266    ipByteBuf.flip();  // Prepare to re-read it back as an int.
267    localIpAddr = ipByteBuf.getInt();
268  }

270  // As ctRcv and ctSrc connections are created, assign them simple incrementing connection IDs.
271  // THREAD: ctrlr
272  int nextConnId() {
273    int connId;
274    synchronized (this) {
275      connId = nextConnId;
276      nextConnId ++;
277    }
278    return connId;
279  }

281  // Utility to get an exception's error message and stack trace as a string.
282  static String exDetails(Exception e) {
283    // Put stack trace into string writer "sw".
284    StringWriter sw = new StringWriter();
285    PrintWriter pw = new PrintWriter(sw);
286    e.printStackTrace(pw);
287    // Return error message and stack trace.
288    return sw.toString();
289  }

292  // Create the UM receiver for reception of handshake messages sent to the source.
293  // Since these messages are sent as UIMs, disable all for this receiver.
294  // THREAD: user
295  private void srcHandshakeRcvCreate() throws Exception {
296    LBMReceiverAttributes rcvAttr = new LBMReceiverAttributes();
297    rcvAttr.setProperty("resolver_query_minimum_initial_interval", "0");
298    rcvAttr.setProperty("resolver_query_maximum_initial_interval", "0");
299    rcvAttr.setProperty("resolver_query_sustain_interval", "0");
300    rcvAttr.setProperty("resolution_number_of_sources_query_threshold", "0");
301    LBMTopic topicObj = ctx.lookupTopic(LbmCt.HANDSHAKE_TOPIC_STR, rcvAttr);
302
303    LBMReceiverCallback srcSideMsgRcvCb = new SrcSideMsgRcvCb(this, ctrlr);
304    umHandshakeRcv = new LBMReceiver(ctx, topicObj, srcSideMsgRcvCb, null);
305    rcvAttr.dispose();
306  }

308  /**
309   * Implementation of {@code LBMReceiver} callback interface for UM receiver events.
310   */
311  private static class SrcSideMsgRcvCb implements LBMReceiverCallback {
312    LbmCt ct;
313    LbmCtCtrlr ctrlr;
314
315    private SrcSideMsgRcvCb(LbmCt ct, LbmCtCtrlr ctrlr) {
316      this.ct = ct;
317      this.ctrlr = ctrlr;
318    }
319
320    // Not part of public API.
321    // THREAD: ctx
322    public int onReceive(Object cbArgs, LBMMessage umMsg) {
323      try {
324        ct.srcHandshake(umMsg);
325      } catch (Exception e) {
326        LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCt::onReceive: ct.srcHandshake: " + LbmCt.exDetails(e) + "\n");
327      }
328      umMsg.dispose();  // The message data was copied out of msg, so the msg can be disposed (for ZOD).
329      return 0;
330    }
331  }

333  // The code below conceptually belongs with LbmCtSrc, but first we need to determine
334  // which instance this message belongs to (or a new instance).
335  // THREAD: ctx
336  private void srcHandshake(LBMMessage umMsg) {
337    if (umMsg.type() == LBM.MSG_DATA) {
338      LbmCtCtrlrCmd cmd = ctrlr.cmdGet();
339      cmd.setSrcHandshake(this, umMsg);  // Copy the message data out of msg.
340      ctrlr.submitNowait(cmd);  // This "calls" cmdSrcHandshake below.
341    }
342  }
343
344  // THREAD: ctrlr
345  boolean cmdSrcHandshake(LbmCtCtrlrCmd cmd) throws Exception {
346    ByteBuffer msgData = cmd.getMsgData();
347    int handshakeType = handshakeParser.parse(msgData);
348    switch (handshakeType) {
349      case LbmCtHandshakeParser.MSG_TYPE_CREQ:
350        srcHandshakeCreq(handshakeParser);
351        break;
352      case LbmCtHandshakeParser.MSG_TYPE_COK:
353        srcHandshakeCok(handshakeParser);
354        break;
355      case LbmCtHandshakeParser.MSG_TYPE_DREQ:
356        srcHandshakeDreq(handshakeParser);
357        break;
358      case LbmCtHandshakeParser.MSG_TYPE_DOK:
359        srcHandshakeDok(handshakeParser);
360        break;
361      default:
362        throw (new LBMException("ct source got unexpected handshake " + handshakeType));
363    }
364    handshakeParser.clear();
365    return true;
366  }

368  // THREAD: ctrlr
369  private void srcHandshakeCreq(LbmCtHandshakeParser handshakeParser) throws Exception {
370    LbmCtSrcConn srcConn;
371
372    if (srcConnMap.containsKey(handshakeParser.getRcvConnKey())) {
373      srcConn = srcConnMap.get(handshakeParser.getRcvConnKey());
374      if (srcConn.getCtSrc().isClosing()) {
375        throw new LBMException("Got creq on exiting src");
376      }
377    } else { // No src connection yet, create one.
378      // Find ctSrc object for this topic.
379      String topicStr = handshakeParser.getTopicStr();
380      if (! ctSrcMap.containsKey(topicStr)) {
381        throw new LBMException("CREQ recvd for topic '" + topicStr + "' with no src");
382      }
383      LbmCtSrc ctSrc = ctSrcMap.get(topicStr);
384      if (ctSrc.isClosing()) {
385        throw new LBMException("Got creq on exiting src");
386      }
387
388      // Create srcConn object and register it.
389      srcConn = new LbmCtSrcConn(ctSrc);
390      srcConn.start(handshakeParser);
391
392      addToSrcConnMap(srcConn);
393    }
394
395    // Have srcConn.  Continue handling CREQ message from inside the conn object.
396    srcConn.handleCreq(handshakeParser);
397  }
398
399  // THREAD: ctrlr
400  private void srcHandshakeCok(LbmCtHandshakeParser handshakeParser) throws Exception {
401    LbmCtSrcConn srcConn;
402
403    if (srcConnMap.containsKey(handshakeParser.getRcvConnKey())) {
404      srcConn = srcConnMap.get(handshakeParser.getRcvConnKey());
405      // Have srcConn.  Continue handling COK message from inside the conn object.
406      srcConn.handleCok(handshakeParser);
407    } else {
408      LBMPubLog.pubLog(LBM.LOG_INFO, "LbtCt::srcHandshakeCok: connection not found  for COK handshake received on topic " + handshakeParser.getTopicStr() + "\n");
409    }
410  }
