024import java.util.*;
025import com.latencybusters.lbm.*;
026
027/**
028 * A connected source.
029 * Maintains an underlying UM source, which is available with the {@link #getUmSrc()} method.
030 * When an {@code LbmCtSrc} is created, its full initialization is deferred until its {@link #start} method is called.
031 */
032@SuppressWarnings("WeakerAccess")  // public API.
033public class LbmCtSrc {
034  private LbmCt ct = null;
035  private LbmCtCtrlr ctrlr = null;
036  private LBMContext ctx = null;
037  private LBMSource umSrc = null;
038  private String topicStr = null;
039  private LbmCtSrcConnCreateCallback connCreateCb = null;
040  private LbmCtSrcConnDeleteCallback connDeleteCb = null;
041  private Object cbArg = null;
042  private Set<LbmCtSrcConn> srcConnSet = null;
043  private int numActiveConns = 0;
044
045  private boolean stopping = false;
046  private LbmCtCtrlrCmd pendingStopCmd = null;
047  private boolean umSourceStopSubmitted = false;
048  private boolean finalStopSubmitted = false;
049
050  /**
051   * Creates a Connected Topic Source object.
052   * Creation of a CT Source object creates an underlying UM source.
053   * Unlike a UM source, creation a CT Source object is not a two step process.
054   * (A UM source requires first allocating a UM topic object, and then creating the source.
055   * A CT source combines the two steps in its {@link #start} method.)
056   * Note that to send a message, the application must access the underlying UM source
057   * with the {@link #getUmSrc()} method.
058   * <p>
059   * This constructor only creates the object.
060   * Its full initialization is deferred until its {@code start} method is called.
061   */
062  public LbmCtSrc() { }
063
064  // Getters.
065  LbmCt getCt() { return ct; }
066  String getTopicStr() { return topicStr; }
067  LbmCtSrcConnCreateCallback getConnCreateCb() { return connCreateCb; }
068  LbmCtSrcConnDeleteCallback getConnDeleteCb() { return connDeleteCb; }
069  Object getCbArg() { return cbArg; }
070  boolean isStopping() { return stopping; }
071
072  /**
073   * Obtain a reference to a CT Source's underlying UM source.
074   * This is needed so that the application can send messages.
075   * @return  reference to underlying UM source.
076   */
077  @SuppressWarnings("WeakerAccess")  // public API.
078  public LBMSource getUmSrc() { return umSrc; }
079
080  /**
081   * Initializes a CT Source object.
082   * This is typically called immediately after the object is created ({@link #LbmCtSrc} constructor).
083   * The {@code start} method can return before any connections are made.
084   * Any messages sent to the source prior to the first connection event may not be received by any receivers.
085   * <p>
086   * When the application is finished using this CT Source, it should be stopped ({@link #stop} API).
087   * <p>
088   * @param inCt  CT object to associate with this CT Source.
089   * @param topicStr  Topic string.
090   *     This is the same as the {@code symbol} parameter when allocating a UM source topic object with
091   *     <a href="https://ultramessaging.github.io/currdoc/doc/JavaAPI/classcom_1_1latencybusters_1_1lbm_1_1LBMTopic.html#ab728a34e695bfccc73c8fa1a33b0e70d">LBMTopic</a>.
092   * @param srcAttr  Attributes object used to configure the underlying UM source.
093   *     This is the same as the {@code lbmsattr} parameter when allocating a UM source topic object with
094   *     <a href="https://ultramessaging.github.io/currdoc/doc/JavaAPI/classcom_1_1latencybusters_1_1lbm_1_1LBMTopic.html#ab728a34e695bfccc73c8fa1a33b0e70d">LBMTopic</a>.
095   * @param srcCb  An object implementing the
096   *     <a href="https://ultramessaging.github.io/currdoc/doc/JavaAPI/interfacecom_1_1latencybusters_1_1lbm_1_1LBMSourceEventCallback.html">LBMSourceEventCallback</a>
097   *     interface.
098   *     This is the same as the {@code cb} parameter when creating a UM source with
099   *     <a href="https://ultramessaging.github.io/currdoc/doc/JavaAPI/classcom_1_1latencybusters_1_1lbm_1_1LBMSource.html#ab8e3998370398ea2929bfaf814d79457">LBMSource</a>.
100   *     It is used to deliver UM source events to the application.
101   *     This parameter is optional (null should be passed if not needed).
102   *     Note that the {@code SRC_EVENT_CONNECT} and {@code SRC_EVENT_DISCONNECT} events should be ignored.
103   *     The CT connection create and delete callbacks should be used instead.
104   * @param connCreateCb  Callback object invoked when a CT Receiver connects to this CT source.
105   *     Technically this parameter is optional (null should be passed if not needed), but its use is central to the
106   *     Connected Topics paradigm.
107   * @param connDeleteCb  Callback object invoked when a CT receiver disconnects from this CT source.
108   *     Technically this parameter is optional (null should be passed if not needed), but its use is central to the
109   *     Connected Topics paradigm.
110   * @param cbArg  Application-specific object to be associated with the connected source, which is passed to the three
111   *     callbacks ({@code srcCb}, {@code connCreateCb}, and {@code connDeleteCb}).
112   *     Note that the CT Source maintains a reference to the cbArg object until the CT Source is stopped.
113   *     This parameter is optional (null should be passed if not needed).
114   * @throws Exception  LBMException thrown.
115   */
116  public void start(LbmCt inCt, String topicStr, LBMSourceAttributes srcAttr, LBMSourceEventCallback srcCb,
117                    LbmCtSrcConnCreateCallback connCreateCb, LbmCtSrcConnDeleteCallback connDeleteCb, Object cbArg)
118      throws Exception
119  {
120    ct = inCt;
121    ctrlr = ct.getCtrlr();
122    ctx = ct.getCtx();
123
124    LbmCtCtrlrCmd nextCmd = ctrlr.cmdGet();
125    nextCmd.setCtSrcStart(this, topicStr, srcAttr, srcCb, connCreateCb, connDeleteCb, cbArg);
126    ctrlr.submitWait(nextCmd);  // This "calls" cmdCtSrcStart below.
127
128    Exception e = nextCmd.getE();  // Save exception to be re-thrown after command free.
129    ctrlr.cmdFree(nextCmd);  // Return command object to free pool.
130    if (e != null) {
131      throw(new Exception(e));
132    }
133  }
134
135  // THREAD: ctrlr
136  boolean cmdCtSrcStart(LbmCtCtrlrCmd cmd) throws Exception {
137    topicStr = cmd.getSrcTopicStr();
138    LBMSourceEventCallback srcCb = cmd.getSrcCb();
139    connCreateCb = cmd.getSrcConnCreateCb();
140    connDeleteCb = cmd.getSrcConnDeleteCb();
141    cbArg = cmd.getSrcCbArg();
142    srcConnSet = new HashSet<>();
143
144    ct.addToCtSrcSet(this);
145
146    LBMTopic topicObj = ctx.allocTopic(topicStr, cmd.getSrcAttr());
147    if (srcCb == null) {
148      umSrc = new LBMSource(ctx, topicObj);
149    } else {
150      umSrc = new LBMSource(ctx, topicObj, srcCb, cbArg);
151    }
152
153    ct.addToSrcMap(this);
154    return true;
155  }
156
157  /**
158   * Stops this CT source object.
159   * Calling this method performs a graceful disconnection of any active connections, blocking the caller until
160   * the disconnections are completed.
161   * <p>
162   * @throws Exception  LBMException thrown.
163   */
164  @SuppressWarnings("WeakerAccess")  // public API.
165  public void stop() throws Exception {
166    LbmCtCtrlr ctrlr = this.ctrlr;  // get a local copy since the command will delete the ctSrc.
167
168    LbmCtCtrlrCmd nextCmd = ctrlr.cmdGet();
169    nextCmd.setCtSrcStop(this);
170    ctrlr.submitWait(nextCmd);  // This "calls" cmdCtSrcStop below.
171
172    Exception e = nextCmd.getE();  // Save exception to be re-thrown after cmd free.
173    ctrlr.cmdFree(nextCmd);  // Return command object to free pool.
174    if (e != null) {
175      throw(new Exception(e));
176    }
177
178    // No longer need to "find" this conn as UIM handshakes are received.
179    ct.removeFromSrcMap(this);
180
181    // Release any references to other objects.
182    clear();
183  }
184
185  // THREAD: ctrlr
186  boolean cmdCtSrcStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) throws Exception {
187    stopping = true;
188    pendingStopCmd = cmd;
189
190    for (LbmCtSrcConn srcConn : srcConnSet) {
191      // These disconnects are submitted to the command queue.
192      srcConn.disconnect();
193    }
194
195    if ((numActiveConns == 0) && (!umSourceStopSubmitted)) {
196      // Since stopping is true, no new connections should become active.
197      // There might be some connection-related commands queued.  Queue the UM source stop behind them.
198      LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
199      nextCmd.setCtSrcUmSourceStop(this);
200      ctrlr.submitNowait(nextCmd);
201      umSourceStopSubmitted = true;
202    }
203
204    return false;
205  }
206
207  // This is called when there are no more "active" connections.
208  // There can still be some connection-related commands queued.
209  // THREAD: ctrlr
210  boolean cmdCtSrcUmSourceStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) throws Exception {
211    if (umSrc != null) {
212      umSrc.close();
213      umSrc = null;
214    }
215
216    // At this point, no new connections will be created.  But there might be some commands in the
217    // queue waiting to be processed.  Submit the final stop to be behind those.
218    if ((numActiveConns != 0) || (! srcConnSet.isEmpty())) {
219      // This should never happen.  Log warning and continue.
220      LBMPubLog.pubLog(LBM.LOG_WARNING, "Internal error: cmdCtSrcUmSourceStop: numActiveConns=" + numActiveConns + " (should be 0), srcConnSet.isEmpty=" + srcConnSet.isEmpty() + " (should be true)\n");
221    } else {
222      if (!finalStopSubmitted) {  // Don't submit another one if one is already pending.
223        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
224        nextCmd.setCtSrcFinalStop(this);
225        ctrlr.submitNowait(nextCmd);  // This "calls" cmdSrcConnStop below.
226
227        finalStopSubmitted = true;
228      }
229    }
230    return true;
231  }
232
233  // There should be no more connections of any kind.
234  // THREAD: ctrlr
235  boolean cmdCtSrcFinalStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) {
236    synchronized (this) {
237      // There should be no connections left at all.
238      try {
239        if (! srcConnSet.isEmpty()) {
240          throw new LBMException("Internal error, cmdCtSrcFinalStop: srcConnSet not empty");
241        }
242      } catch (Exception e) {
243        // Pass the exception back to the application.
244        pendingStopCmd.setE(e);
245      }
246
247      ct.removeFromCtSrcSet(this);
248      // This ct source is now fully removed from the LbmCt.
249
250      // Wake up ctSrc stop().
251      ctrlr.cmdComplete(pendingStopCmd);
252    }
253    return true;
254  }
255
256  // Release references to other objects.  This is in case the
257  // application is holding a reference to the CT source.
258  private void clear() {
259    ct = null;
260    ctrlr = null;
261    ctx = null;
262    umSrc = null;
263    topicStr = null;
264    connCreateCb = null;
265    connDeleteCb = null;
266    cbArg = null;
267    srcConnSet = null;
268    pendingStopCmd = null;
269  }
270
271  void addToSrcConnSet(LbmCtSrcConn srcConn) {
272    srcConnSet.add(srcConn);
273  }
274
275  void removeFromSrcConnSet(LbmCtSrcConn srcConn) {
276    srcConnSet.remove(srcConn);
277    if (isStopping() && srcConnSet.isEmpty()) {
278      if (!umSourceStopSubmitted) {
279        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
280        nextCmd.setCtSrcUmSourceStop(this);
281        ctrlr.submitNowait(nextCmd);
282
283        umSourceStopSubmitted = true;
284      }
285      else if (!finalStopSubmitted) {
286        // At this point, no new connections will be created.  But there might be some commands in the
287        // queue waiting to be processed.  Submit the final stop to be behind those.
288        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
289        nextCmd.setCtSrcFinalStop(this);
290        ctrlr.submitNowait(nextCmd);
291
292        finalStopSubmitted = true;
293      }
294    }
295  }
296
297  void incrementActiveConns() {
298    numActiveConns++;
299  }
300
301  void decrementActiveConns() {
302    numActiveConns--;
303    if (isStopping()) {
304      if ((numActiveConns == 0) && (!umSourceStopSubmitted)) {
305        // There might be some connection-related commands queued.  Queue the UM receiver stop behind.
306        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
307        nextCmd.setCtSrcUmSourceStop(this);
308        ctrlr.submitNowait(nextCmd);
309        umSourceStopSubmitted = true;
310      }
311    }
312  }
