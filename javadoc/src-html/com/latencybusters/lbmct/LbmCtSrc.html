024import java.util.*;
025import com.latencybusters.lbm.*;
026
027@SuppressWarnings("WeakerAccess")  // public API.
028public class LbmCtSrc {
029  private LbmCt ct = null;
030  private LbmCtCtrlr ctrlr = null;
031  private LBMContext ctx = null;
032  private LBMSource umSrc = null;
033  private String topicStr = null;
034  private LbmCtSrcConnCreateCallback connCreateCb = null;
035  private LbmCtSrcConnDeleteCallback connDeleteCb = null;
036  private Object cbArg = null;
037  private Set<LbmCtSrcConn> srcConnSet = null;
038  private int numActiveConns = 0;
039
040  private boolean closing = false;
041  private LbmCtCtrlrCmd pendingCloseCmd = null;
042  private boolean umSourceCloseSubmitted = false;
043  private boolean finalCloseSubmitted = false;
044
045  // Getters.
046  LbmCt getCt() { return ct; }
047  String getTopicStr() { return topicStr; }
048  LbmCtSrcConnCreateCallback getConnCreateCb() { return connCreateCb; }
049  LbmCtSrcConnDeleteCallback getConnDeleteCb() { return connDeleteCb; }
050  Object getCbArg() { return cbArg; }
051  boolean isClosing() { return closing; }
052
053  // Public API to get underlying UM source object from CT source.
054  @SuppressWarnings("WeakerAccess")  // public API.
055  public LBMSource getUmSrc() { return umSrc; }
056
057  // Public API for actual creation of the UM source.  Just hand off to ctrlr thread.
058  @SuppressWarnings("WeakerAccess")  // public API.
059  public void start(LbmCt inCt, String topicStr, LBMSourceAttributes srcAttr, LBMSourceEventCallback srcCb,
060                    LbmCtSrcConnCreateCallback connCreateCb, LbmCtSrcConnDeleteCallback connDeleteCb, Object cbArg)
061      throws Exception
062  {
063    ct = inCt;
064    ctrlr = ct.getCtrlr();
065    ctx = ct.getCtx();
066
067    LbmCtCtrlrCmd nextCmd = ctrlr.cmdGet();
068    nextCmd.setCtSrcStart(this, topicStr, srcAttr, srcCb, connCreateCb, connDeleteCb, cbArg);
069    ctrlr.submitWait(nextCmd);  // This "calls" cmdCtSrcStart below.
070
071    Exception e = nextCmd.getE();  // Save exception to be re-thrown after command free.
072    ctrlr.cmdFree(nextCmd);  // Return command object to free pool.
073    if (e != null) {
074      throw(new Exception(e));
075    }
076  }
077
078  // THREAD: ctrlr
079  boolean cmdCtSrcStart(LbmCtCtrlrCmd cmd) throws Exception {
080    topicStr = cmd.getSrcTopicStr();
081    LBMSourceEventCallback srcCb = cmd.getSrcCb();
082    connCreateCb = cmd.getSrcConnCreateCb();
083    connDeleteCb = cmd.getSrcConnDeleteCb();
084    cbArg = cmd.getSrcCbArg();
085    srcConnSet = new HashSet<>();
086
087    ct.addToCtSrcSet(this);
088
089    LBMTopic topicObj = ctx.allocTopic(topicStr, cmd.getSrcAttr());
090    if (srcCb == null) {
091      umSrc = new LBMSource(ctx, topicObj);
092    } else {
093      umSrc = new LBMSource(ctx, topicObj, srcCb, cbArg);
094    }
095
096    ct.addToSrcMap(this);
097    return true;
098  }
099
100  // Public API for deleting CT source.
101  @SuppressWarnings("WeakerAccess")  // public API.
102  public void close() throws Exception {
103    LbmCtCtrlr ctrlr = this.ctrlr;  // get a local copy since the command will delete the ctSrc.
104
105    LbmCtCtrlrCmd nextCmd = ctrlr.cmdGet();
106    nextCmd.setCtSrcClose(this);
107    ctrlr.submitWait(nextCmd);  // This "calls" cmdCtSrcClose below.
108
109    Exception e = nextCmd.getE();  // Save exception to be re-thrown after cmd free.
110    ctrlr.cmdFree(nextCmd);  // Return command object to free pool.
111    if (e != null) {
112      throw(new Exception(e));
113    }
114
115    // No longer need to "find" this conn as UIM handshakes are received.
116    ct.removeFromSrcMap(this);
117
118    // Release any references to other objects.
119    clear();
120  }
121
122  // THREAD: ctrlr
123  boolean cmdCtSrcClose(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) throws Exception {
124    closing = true;
125    pendingCloseCmd = cmd;
126
127    for (LbmCtSrcConn srcConn : srcConnSet) {
128      // These disconnects are submitted to the command queue.
129      srcConn.disconnect();
130    }
131
132    if ((numActiveConns == 0) && (!umSourceCloseSubmitted)) {
133      // Since closing is true, no new connections should become active.
134      // There might be some connection-related commands queued.  Queue the UM source close behind them.
135      LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
136      nextCmd.setCtSrcUmSourceClose(this);
137      ctrlr.submitNowait(nextCmd);
138      umSourceCloseSubmitted = true;
139    }
140
141    return false;  // Command completion is triggered by finalCloseMaybe().
142  }
143
144  // This is called when there are no more "active" connections.
145  // There can still be some connection-related commands queued.
146  // THREAD: ctrlr
147  boolean cmdCtSrcUmSourceClose(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) throws Exception {
148    if (umSrc != null) {
149      umSrc.close();
150      umSrc = null;
151    }
152
153    // At this point, no new connections will be created.  But there might be some commands in the
154    // queue waiting to be processed.  Submit the final close to be behind those.
155    if ((numActiveConns != 0) || (! srcConnSet.isEmpty())) {
156      // This should never happen.  Log warning and continue.
157      LBMPubLog.pubLog(LBM.LOG_WARNING, "Internal error: cmdCtSrcUmSourceClose: numActiveConns=" + numActiveConns + " (should be 0), srcConnSet.isEmpty=" + srcConnSet.isEmpty() + " (should be true)\n");
158    } else {
159      if (!finalCloseSubmitted) {  // Don't submit another one if one is already pending.
160        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
161        nextCmd.setCtSrcFinalClose(this);
162        ctrlr.submitNowait(nextCmd);  // This "calls" cmdSrcConnClose below.
163
164        finalCloseSubmitted = true;
165      }
166    }
167    return true;
168  }
169
170  // There should be no more connections of any kind.
171  // THREAD: ctrlr
172  boolean cmdCtSrcFinalClose(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) {
173    synchronized (this) {
174      // There should be no connections left at all.
175      try {
176        if (! srcConnSet.isEmpty()) {
177          throw new LBMException("Internal error, cmdCtSrcFinalClose: srcConnSet not empty");
178        }
179      } catch (Exception e) {
180        // Pass the exception back to the application.
181        pendingCloseCmd.setE(e);
182      }
183
184      ct.removeFromCtSrcSet(this);
185      // This ct source is now fully removed from the LbmCt.
186
187      // Wake up ctSrc close().
188      ctrlr.cmdComplete(pendingCloseCmd);
189    }
190    return true;
191  }
192
193  // Release references to other objects.  This is in case the
194  // application is holding a reference to the CT source.
195  private void clear() {
196    ct = null;
197    ctrlr = null;
198    ctx = null;
199    umSrc = null;
200    topicStr = null;
201    connCreateCb = null;
202    connDeleteCb = null;
203    cbArg = null;
204    srcConnSet = null;
205    pendingCloseCmd = null;
206  }
207
208  void addToSrcConnSet(LbmCtSrcConn srcConn) {
209    srcConnSet.add(srcConn);
210  }
211
212  void removeFromSrcConnSet(LbmCtSrcConn srcConn) {
213    srcConnSet.remove(srcConn);
214    if (isClosing() && srcConnSet.isEmpty()) {
215      if (!umSourceCloseSubmitted) {
216        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
217        nextCmd.setCtSrcUmSourceClose(this);
218        ctrlr.submitNowait(nextCmd);
219
220        umSourceCloseSubmitted = true;
221      }
222      else if (!finalCloseSubmitted) {
223        // At this point, no new connections will be created.  But there might be some commands in the
224        // queue waiting to be processed.  Submit the final close to be behind those.
225        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
226        nextCmd.setCtSrcFinalClose(this);
227        ctrlr.submitNowait(nextCmd);
228
229        finalCloseSubmitted = true;
230      }
231    }
232  }
233
234  void incrementActiveConns() {
235    numActiveConns++;
236  }
237
238  void decrementActiveConns() {
239    numActiveConns--;
240    if (isClosing()) {
241      if ((numActiveConns == 0) && (!umSourceCloseSubmitted)) {
242        // There might be some connection-related commands queued.  Queue the UM receiver close behind.
243        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
244        nextCmd.setCtSrcUmSourceClose(this);
245        ctrlr.submitNowait(nextCmd);
246        umSourceCloseSubmitted = true;
247      }
248    }
249  }
250}
