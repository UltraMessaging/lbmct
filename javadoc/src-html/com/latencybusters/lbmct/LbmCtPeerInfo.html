024import java.nio.*;
025import com.latencybusters.lbm.*;
026
027/**
028 * A passive object that provides useful information to a Connected Topics application.
029 * The object is offered to the application at the connect and disconnect events, as delivered by
030 * {@link}
031 * There are multiple fields of potential interest in the object, and they are not all valid at all times (for
032 * example, the "end" sequence number is not known when the connection is first made).
033 * The {@link #getFlags} method allows
034 */
035@SuppressWarnings("WeakerAccess")  // public API.
036public class LbmCtPeerInfo {
037  @SuppressWarnings("WeakerAccess")  // public API.
038  public final static int STATUS_OK = 0;
039  @SuppressWarnings("WeakerAccess")  // public API.
040  public final static int STATUS_BAD_STOP = -1;
041
042
043  // Flag bits for "flags" field.
044  @SuppressWarnings("WeakerAccess")  // public API.
045  public final static int FLAGS_SRC_METADATA      = 0x01;
046  @SuppressWarnings("WeakerAccess")  // public API.
047  public final static int FLAGS_RCV_METADATA      = 0x02;
048  @SuppressWarnings("WeakerAccess")  // public API.
049  public final static int FLAGS_RCV_SOURCE_STR    = 0x04;
050  @SuppressWarnings("WeakerAccess")  // public API.
051  public final static int FLAGS_RCV_START_SEQ_NUM = 0x08;
052  @SuppressWarnings("WeakerAccess")  // public API.
053  public final static int FLAGS_RCV_END_SEQ_NUM   = 0x10;
054
055  private int status = STATUS_BAD_STOP;
056  private int flags = 0;
057  private ByteBuffer srcMetadata = null;
058  private ByteBuffer rcvMetadata = null;
059  private long rcvStartSequenceNumber = 0;
060  private long rcvEndSequenceNumber = 0;
061  private String rcvSourceStr = null;
062
063  // No constructor needed.
064
065  @SuppressWarnings("WeakerAccess")  // public API.
066  public int getStatus() {
067    return status;
068  }
069  @SuppressWarnings("WeakerAccess")  // public API.
070  public int getFlags() { return flags; }
071  @SuppressWarnings("WeakerAccess")  // public API.
072  public ByteBuffer getSrcMetadata() throws Exception {
073    if ((flags & FLAGS_SRC_METADATA) == 0) {
074      throw new LBMException("srcMetadata undefined");
075    }
076    return srcMetadata;
077  }
078  @SuppressWarnings("WeakerAccess")  // public API.
079  public ByteBuffer getRcvMetadata() throws Exception {
080    if ((flags & FLAGS_RCV_METADATA) == 0) {
081      throw new LBMException("rcvMetadata undefined");
082    }
083    return rcvMetadata;
084  }
085  @SuppressWarnings("WeakerAccess")  // public API.
086  public long getRcvStartSequenceNumber() throws Exception {
087    if ((flags & FLAGS_RCV_START_SEQ_NUM) == 0) {
088      throw new LBMException("rcvStartSequenceNumber undefined");
089    }
090    return rcvStartSequenceNumber;
091  }
092  @SuppressWarnings("WeakerAccess")  // public API.
093  public long getRcvEndSequenceNumber() throws Exception {
094    if ((flags & FLAGS_RCV_END_SEQ_NUM) == 0) {
095      throw new LBMException("rcvEndSequenceNumber undefined");
096    }
097    return rcvEndSequenceNumber;
098  }
099  @SuppressWarnings("WeakerAccess")  // public API.
100  public String getRcvSourceStr() throws Exception {
101    if ((flags & FLAGS_RCV_SOURCE_STR) == 0) {
102      throw new LBMException("rcvSourceStr undefined");
103    }
104    return rcvSourceStr;
105  }
106
107  void setStatus(int status) { this.status = status; }
108
109  void setSrcMetadata(ByteBuffer inMetadata) {
110    // Make a deep copy of the metadata.
111    int oldPosition = inMetadata.position();
112    int len = inMetadata.remaining();
113    srcMetadata = ByteBuffer.allocate(len);
114    inMetadata.get(srcMetadata.array(), 0, len);
115    srcMetadata.position(len);
116    srcMetadata.flip();
117
118    inMetadata.position(oldPosition);  // Restore input metadata byte buffer.
119    flags = flags | FLAGS_SRC_METADATA;
120  }
121  void setRcvMetadata(ByteBuffer inMetadata) {
122    // Make a deep copy of the metadata.
123    int oldPosition = inMetadata.position();
124    int len = inMetadata.limit();
125    rcvMetadata = ByteBuffer.allocate(len);
126    inMetadata.get(rcvMetadata.array(), 0, len);
127    rcvMetadata.position(len);
128    rcvMetadata.flip();
129
130    inMetadata.position(oldPosition);  // Restore input metadata byte buffer.
131    flags = flags | FLAGS_RCV_METADATA;
132  }
133  void setRcvStartSequenceNumber(long inStartSequenceNumber) {
134    rcvStartSequenceNumber = inStartSequenceNumber;
135    flags = flags | FLAGS_RCV_START_SEQ_NUM;
136  }
137  void setRcvEndSequenceNumber(long inEndSequenceNumber) {
138    rcvEndSequenceNumber = inEndSequenceNumber;
139    flags = flags | FLAGS_RCV_END_SEQ_NUM;
140  }
141  void setRcvSourceStr(String inSourceStr) {
142    rcvSourceStr = inSourceStr;
143    flags = flags | FLAGS_RCV_SOURCE_STR;
144  }
145}
