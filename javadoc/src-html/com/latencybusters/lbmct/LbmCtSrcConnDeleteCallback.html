024/**
025 * Callback interface to deliver the "connection deleted" event to a source application.
026 * The application supplies an implementation of this interface to {@link LbmCtSrc#start}.
027 */
028public interface LbmCtSrcConnDeleteCallback {
029  /**
030   * Source application callback method, called by CT, to indicate that a connection is deleted.
031   * <p>
032   * @param src  The CT Source object that the deleted connection applies to.
033   * @param peerInfo  Useful information about the connection, such as the metadata of the connected receiver.
034   * @param cbArg  Application-specific object supplied by the application to {@link LbmCtSrc#start}.
035   * @param connCbArg  Application-specific object associated with the connection, supplied by the application as
036   *     the return value from {@link LbmCtSrcConnCreateCallback#onSrcConnCreate}.
037   */
038  void onSrcConnDelete(LbmCtSrc src, LbmCtPeerInfo peerInfo, Object cbArg, Object connCbArg);
039}  // LbmCtSrcConnDeleteCallback
