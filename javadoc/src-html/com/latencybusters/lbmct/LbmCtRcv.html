024import java.util.*;
025import com.latencybusters.lbm.*;
026
027@SuppressWarnings("WeakerAccess")  // public API.
028public class LbmCtRcv {
029  private Object lock = new Object();
030  private LbmCt ct = null;
031  private LbmCtCtrlr ctrlr = null;
032  private LBMContext ctx = null;
033  private LBMReceiver umRcv = null;
034  private String topicStr = null;
035  private LBMReceiverCallback rcvCb = null;
036  private LbmCtRcvConnCreateCallback connCreateCb = null;
037  private LbmCtRcvConnDeleteCallback connDeleteCb = null;
038  private Object cbArg = null;
039  private Set<LbmCtRcvConn> rcvConnSet = null;
040  private int numActiveConns = 0;
041
042  private boolean stopping = false;
043  private LbmCtCtrlrCmd pendingStopCmd = null;
044  private boolean umReceiverStopSubmitted = false;
045  private boolean finalStopSubmitted = false;
046
047  // Getters.
048  Object getLock() { return lock; }
049  LbmCt getCt() { return ct; }
050  String getTopicStr() { return topicStr; }
051  LBMReceiverCallback getRcvCb() { return rcvCb; }
052  LbmCtRcvConnCreateCallback getConnCreateCb() { return connCreateCb; }
053  LbmCtRcvConnDeleteCallback getConnDeleteCb() { return connDeleteCb; }
054  Object getCbArg() { return cbArg; }
055  boolean isStopping() { return stopping; }
056
057  // Public API for actual creation of UM receiver.  Just hand off to ctrlr thread.
058  public void start(LbmCt inCt, String topicStr, LBMReceiverAttributes rcvAttr, LBMReceiverCallback rcvCb,
059                    LbmCtRcvConnCreateCallback connCreateCb, LbmCtRcvConnDeleteCallback connDeleteCb, Object cbArg)
060      throws Exception
061  {
062    ct = inCt;
063    ctrlr = ct.getCtrlr();
064    ctx = ct.getCtx();
065    LbmCtCtrlrCmd cmd;
066    try {
067      cmd = ctrlr.cmdGet();
068      cmd.setCtRcvStart(this, topicStr, rcvAttr, rcvCb, connCreateCb, connDeleteCb, cbArg);
069      ctrlr.submitWait(cmd);  // This "calls" cmdCtRcvStart below.
070    } catch (Exception e) {
071      throw (new Exception(e));
072    }
073
074    Exception e = cmd.getE();
075    ctrlr.cmdFree(cmd);  // Return command object to free pool.
076    if (e != null) {
077      throw (new Exception(e));
078    }
079  }
080
081  // THREAD: ctrlr
082  boolean cmdCtRcvStart(LbmCtCtrlrCmd cmd) throws Exception {
083    topicStr = cmd.getRcvTopicStr();
084    connCreateCb = cmd.getRcvConnCreateCb();
085    connDeleteCb = cmd.getRcvConnDeleteCb();
086    rcvCb = cmd.getRcvCb();
087    cbArg = cmd.getRcvCbArg();
088    SrcNotification srcNotification = new SrcNotification(this, ct, ctrlr);
089
090    rcvConnSet = new HashSet<>();
091
092    ct.addToCtRcvSet(this);
093
094    LBMReceiverAttributes rcvAttr = cmd.getRcvAttr();
095    if (rcvAttr == null) {
096      // User didn't supply an attribute object, create our own.
097      rcvAttr = new LBMReceiverAttributes();
098    }
099
100    rcvAttr.setSourceNotificationCallbacks(srcNotification, srcNotification, null);
101
102    LBMTopic topicObj = new LBMTopic(ctx, topicStr, rcvAttr);
103    RcvSideMsgRcvCb rcvSideMsgRcvCb = new RcvSideMsgRcvCb(ct, ctrlr);
104    umRcv = new LBMReceiver(ctx, topicObj, rcvSideMsgRcvCb, cbArg);
105
106    if (cmd.getRcvAttr() == null) {
107      // We created our own attribute object; delete it.
108      rcvAttr.dispose();
109    }
110    return true;
111  }
112
113  // Public API for deleting a CT receiver.
114  @SuppressWarnings("WeakerAccess")  // public API.
115  public void stop() throws Exception {
116    LbmCtCtrlr ctrlr = this.ctrlr;  // get a local copy since the command will delete the ctSrc.
117
118    LbmCtCtrlrCmd nextCmd = ctrlr.cmdGet();
119    nextCmd.setCtRcvStop(this);
120    ctrlr.submitWait(nextCmd);  // This "calls" cmdCtRcvStop below.
121
122    Exception e = nextCmd.getE();  // Save exception for re-throwing after freeing the command.
123    ctrlr.cmdFree(nextCmd);  // Return command object to free pool.
124    if (e != null) {
125      throw (new Exception(e));
126    }
127
128    // Release any references to other objects.
129    clear();
130  }
131
132  // THREAD: ctrlr
133  boolean cmdCtRcvStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) {
134    synchronized (this) {
135      stopping = true;
136      pendingStopCmd = cmd;
137
138      for (LbmCtRcvConn rcvConn : rcvConnSet) {
139        // These disconnects are submitted to the command queue.
140        rcvConn.disconnect();
141      }
142
143      if ((numActiveConns == 0) && (!umReceiverStopSubmitted)) {
144        // Since stopping is true, no new connections should become active.
145        // There might be some connection-related commands queued.  Queue the UM receiver stop behind them.
146        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
147        nextCmd.setCtRcvUmReceiverStop(this);
148        ctrlr.submitNowait(nextCmd);
149        umReceiverStopSubmitted = true;
150      }
151    }
152
153    return false;
154  }
155
156  // This is called when there are no more "active" connections.  But there can still be STOP_WAIT connections.
157  // Also, there can be some connection-related commands queued.
158  // THREAD: ctrlr
159  boolean cmdCtRcvUmReceiverStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) throws Exception {
160    // Stop the UM receiver outside of the lock because the call to stop can directly call onSourceDelete(), which
161    // might need to take the lock.
162    if (umRcv != null) {
163      umRcv.close();
164      umRcv = null;
165    }
166
167    // At this point, no new connections will be created.  But there might be some commands in the
168    // queue waiting to be processed.  Submit the final stop to be behind those.
169    synchronized(this) {
170      if (numActiveConns != 0) {
171        // This should never happen.  Log warning and continue.
172        LBMPubLog.pubLog(LBM.LOG_WARNING, "Internal error: cmdCtRcvUmReceiverStop: numActiveConns=" + numActiveConns + "\n");
173      }
174      if (rcvConnSet.isEmpty()) {
175        // No more
176        if (!finalStopSubmitted) {  // Don't submit another one if one is already pending.
177          LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
178          nextCmd.setCtRcvFinalStop(this);
179          ctrlr.submitNowait(nextCmd);  // This "calls" cmdRcvConnStop below.
180
181          finalStopSubmitted = true;
182        }
183      }
184    }
185    return true;
186  }
187
188  // There should be no more connections of any kind, active or STOP_WAIT.
189  // THREAD: ctrlr
190  boolean cmdCtRcvFinalStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) {
191    synchronized (this) {
192      // There should be no connections left at all.
193      try {
194        if (! rcvConnSet.isEmpty()) {
195          throw new LBMException("Internal error, cmdCtRcvFinalStop: rcvConnSet not empty");
196        }
197      } catch (Exception e) {
198        // Pass the exception back to the application.
199        pendingStopCmd.setE(e);
200      }
201
202      ct.removeFromCtRcvSet(this);
203      // This ct receiver is now fully removed from the LbmCt.
204
205      // Wake up ctRcv stop().
206      ctrlr.cmdComplete(pendingStopCmd);
207    }
208    return true;
209  }
210
211  // Release references to other objects.  This is in case the
212  // application is holding a reference to the CT receiver.
213  private void clear() {
214    ct = null;
215    ctrlr = null;
216    ctx = null;
217    umRcv = null;
218    topicStr = null;
219    rcvCb = null;
220    connCreateCb = null;
221    connDeleteCb = null;
222    cbArg = null;
223    rcvConnSet = null;
224    pendingStopCmd = null;
225  }
226
227  void addToRcvConnSet(LbmCtRcvConn rcvConn) {
228    rcvConnSet.add(rcvConn);
229  }
230
231  void removeFromRcvConnSet(LbmCtRcvConn rcvConn) {
232    synchronized (this) {
233      rcvConnSet.remove(rcvConn);
234      if (isStopping() && rcvConnSet.isEmpty()) {
235        if (!umReceiverStopSubmitted) {
236          LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
237          nextCmd.setCtRcvUmReceiverStop(this);
238          ctrlr.submitNowait(nextCmd);
239
240          umReceiverStopSubmitted = true;
241        }
242        else if (!finalStopSubmitted) {  // Don't submit another one if one is already pending.
243          // At this point, no new connections will be created.  But there might be some commands in the
244          // queue waiting to be processed.  Submit the final stop to be behind those.
245          LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
246          nextCmd.setCtRcvFinalStop(this);
247          ctrlr.submitNowait(nextCmd);
248
249          finalStopSubmitted = true;
250        }
251      }
252    }
253  }
254
255  void incrementActiveConns() {
256    synchronized (this) {
257      numActiveConns++;
258    }
259  }
260
261  void decrementActiveConns() {
262    synchronized (this) {
263      numActiveConns--;
264      if (isStopping()) {
265        if ((numActiveConns == 0) && (!umReceiverStopSubmitted)) {
266          // There might be some connection-related commands queued.  Queue the UM receiver stop behind.
267          LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
268          nextCmd.setCtRcvUmReceiverStop(this);
269          ctrlr.submitNowait(nextCmd);
270          umReceiverStopSubmitted = true;
271        }
272      }
273    }
274  }

276  private static class SrcNotification implements LBMSourceCreationCallback, LBMSourceDeletionCallback {
277    LbmCtRcv ctRcv;
278    LbmCt ct;
279    LbmCtCtrlr ctrlr;
280
281    private SrcNotification(LbmCtRcv ctRcv, LbmCt ct, LbmCtCtrlr ctrlr) {
282      this.ctRcv = ctRcv;
283      this.ct = ct;
284      this.ctrlr = ctrlr;
285    }
286
287    // Not part of public API.
288    // THREAD: ctx
289    public Object onNewSource(String sourceStr, Object cbObj) {
290      LbmCtRcvConn rcvConn = new LbmCtRcvConn(ctRcv);
291
292      try {
293        ct.dbg("onNewSource: " + rcvConn);
294        rcvConn.start(sourceStr);
295      } catch (Exception e) {
296        LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCtRcv:OnNewSource(" + sourceStr + "): " + LbmCt.exDetails(e) + "\n");
297      }
298      // Return the rcvConn object as the per-source clientd.
299      return rcvConn;
300    }
301
302    // Not part of public API.
303    // THREAD: ctx
304    public int onSourceDelete(String source, Object cbObj, Object sourceCbObj) {
305      LbmCtRcvConn rcvConn = (LbmCtRcvConn)sourceCbObj;
306
307      ct.dbg("onSourceDelete: " + rcvConn);
308      try {
309        rcvConn.stop();
310      } catch (Exception e) {
311        LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCtRcv:onSourceDelete: " + LbmCt.exDetails(e) + "\n");
312      }
313      return 0;
314    }
315  }

317  private static class RcvSideMsgRcvCb implements LBMReceiverCallback {
318    LbmCt ct;
319    LbmCtCtrlr ctrlr;
320
321    private RcvSideMsgRcvCb(LbmCt ct, LbmCtCtrlr ctrlr) {
322      this.ct = ct;
323      this.ctrlr = ctrlr;
324    }
325
326    // Not part of public API.
327    // THREAD: ctx
328    public int onReceive(Object cbArgs, LBMMessage umMsg) {
329      // Connected Topics receivers don't use BOS/EOS for anything.
330      if ((umMsg.type() == LBM.MSG_BOS) || (umMsg.type() == LBM.MSG_EOS)) {
331        umMsg.dispose();  // Ignore.
332      } else if (umMsg.sourceClientObject() == null) {
333        LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCtRcv.onReceive: topic=" + umMsg.topicName() + ", sourceClientObject is null\n");
334        umMsg.dispose();
335      } else {
336        try {
337          LbmCtRcvConn rcvConn = (LbmCtRcvConn)umMsg.sourceClientObject();
338          ct.dbg("onReceive: " + rcvConn);
339          // Have rcvConn.  Continue handling inside the conn object.
340          rcvConn.handleMsg(umMsg);
341          // Don't dispose message here because most received messages are for user application, and the app gets to
342          // decide the message disposition.
343        } catch (Exception e) {
344          LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCtRcv.onReceive: " + LbmCt.exDetails(e) + "\n");
345        }
346      }
347      return 0;
348    }
349  }
