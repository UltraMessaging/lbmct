024/**
025 * Container object holding configuration of an instance of {@code LbmCt}.
026 * The caller creates an config instance and sets desired options, and then * passes it to {@link LbmCt#start}.
027 * <p>
028 * Note that the CT object retains a reference to the passed-in configuration object, so it is not recommended to modify
029 * options after the {@code LbmCt} is created, except for testing purposes.
030 * Also, if creating multiple CT objects, it is generally recommended to create multiple configuration object
031 * instances, one for each CT object (unless they should all share exactly the same configuration).
032 *
033 */
034@SuppressWarnings("WeakerAccess")  // public API.
035public class LbmCtConfig {
036  /**
037   * Bit set in {@code testBits} to enable debug event logging.
038   * This bit controls the operation of {@code dbg} and {@code debugQ}.
039   * See <a href="https://ultramessaging.github.io/lbmct/doc/Java_Userguide.html#debugging">Debugging</a>
040   * for information on using this debugging feature.
041   * See {@link #setTestBits}.
042   */
043  @SuppressWarnings("WeakerAccess")  // public API.
044  public final static int TEST_BITS_DEBUG   = 0x00000001;
045  /**
046   * Bit set in {@code testBits} to disable sending of CREQ handshake.
047   * This is to test retry logic.
048   * See {@link #setTestBits}.
049   */
050  @SuppressWarnings("WeakerAccess")  // public API.
051  public final static int TEST_BITS_NO_CREQ = 0x00000002;
052  /**
053   * Bit set in {@code testBits} to disable sending of CRSP handshake.
054   * This is to test retry logic.
055   * See {@link #setTestBits}.
056   */
057  @SuppressWarnings("WeakerAccess")  // public API.
058  public final static int TEST_BITS_NO_CRSP = 0x00000004;
059  /**
060   * Bit set in {@code testBits} to disable sending of COK handshake.
061   * This is to test retry logic.
062   * See {@link #setTestBits}.
063   */
064  @SuppressWarnings("WeakerAccess")  // public API.
065  public final static int TEST_BITS_NO_COK  = 0x00000008;
066  /**
067   * Bit set in {@code testBits} to disable sending of DREQ handshake.
068   * This is to test retry logic.
069   * See {@link #setTestBits}.
070   */
071  @SuppressWarnings("WeakerAccess")  // public API.
072  public final static int TEST_BITS_NO_DREQ = 0x00000010;
073  /**
074   * Bit set in {@code testBits} to disable sending of DRSP handshake.
075   * This is to test retry logic.
076   * See {@link #setTestBits}.
077   */
078  @SuppressWarnings("WeakerAccess")  // public API.
079  public final static int TEST_BITS_NO_DRSP = 0x00000020;
080  /**
081   * Bit set in {@code testBits} to disable sending of DOK handshake.
082   * This is to test retry logic.
083   * See {@link #setTestBits}.
084   */
085  @SuppressWarnings("WeakerAccess")  // public API.
086  public final static int TEST_BITS_NO_DOK  = 0x00000040;
087
088  // Default values for config options.
089  /**
090   * Initial value for {@code testBits}; all testing behavior disabled.
091   * See {@link #setTestBits}.
092   */
093  final static int CT_CONFIG_DEFAULT_TEST_BITS  = 0x00000000;
094  /**
095   * Initial value for {@code domainId}; no UM domain ID used.
096   * See {@link #setDomainId}.
097   */
098  final static int CT_CONFIG_DEFAULT_DOMAIN_ID  = -1;
099  /**
100   * Initial value for {@code delayCreq}; receiver waits this long before
101   * sending CREQ handshake.
102   * The default value should be long enough to allow the transport session to be established, and even in those
103   * cases where more time is needed, the retry logic will provide that time.
104   * See {@link #setDelayCreq}.
105   */
106  final static int CT_CONFIG_DEFAULT_DELAY_CREQ = 10;    // 10 ms
107  /**
108   * Initial value for {@code retryIvl}; wait this long for a handshake
109   * operation before re-trying.
110   * Retries should happen only very rarely.
111   * See {@link #setRetryIvl}.
112   */
113  final static int CT_CONFIG_DEFAULT_RETRY_IVL  = 1000;  // 1 sec
114  /**
115   * Initial value for {@code maxTries}; try a handshake operation this
116   * many times before giving up.
117   * Be aware that if a network outage lasts longer than {@code maxTries} * {@code retryIvl}, but the transport
118   * session does not time out, the connection could fail to be established.
119   * The defaults were chosen to be longer than the default transport session timeouts for LBT-RM and LBT-RU.
120   * See {@link #setMaxTries}.
121   */
122  final static int CT_CONFIG_DEFAULT_MAX_TRIES  = 35;
123  /**
124   * Initial value for {@code preDelivery}; disable delivery of UM messages
125   * to receiver when not connected.
126   * See {@link #setPreDelivery}.
127   */
128  final static int CT_CONFIG_DEFAULT_PRE_DELIVERY = 0;
129
130  // Bit map of options used for testing or debugging
131  private int testBits = CT_CONFIG_DEFAULT_TEST_BITS;
132  private int domainId = CT_CONFIG_DEFAULT_DOMAIN_ID;
133  private int delayCreq = CT_CONFIG_DEFAULT_DELAY_CREQ;
134  private int retryIvl = CT_CONFIG_DEFAULT_RETRY_IVL;
135  private int maxTries = CT_CONFIG_DEFAULT_MAX_TRIES;
136  private int preDelivery = CT_CONFIG_DEFAULT_PRE_DELIVERY;
137
138  /**
139   * Constructor for LbmCt configuration object, initialized to default values.
140   * Use setters to override defaults.
141   */
142  @SuppressWarnings("WeakerAccess")  // public API.
143  public LbmCtConfig() {
144  }
145
146  /**
147   * Override the default value for the configuration object's test bits.
148   * See {@link #CT_CONFIG_DEFAULT_TEST_BITS}.
149   * <p>
150   * @param testBits  Integer bit map of debug behaviors to enable.
151   *     One or more of the following ORed together:
152   *     {@link #TEST_BITS_DEBUG}, {@link #TEST_BITS_NO_CREQ}, {@link #TEST_BITS_NO_CRSP}, {@link #TEST_BITS_NO_COK},
153   *     {@link #TEST_BITS_NO_DREQ}, {@link #TEST_BITS_NO_DRSP}, {@link #TEST_BITS_NO_DOK}.
154   */
155  @SuppressWarnings("WeakerAccess")  // public API.
156  public void setTestBits(int testBits) { this.testBits = testBits; }
157
158  /**
159   * Override the default value for the configuration object's Topic Resolution Domain ID.
160   * If not set (defaults to -1), the CT layer does not assume a domain ID, and will rely on the
161   * <a href="https://ultramessaging.github.io/currdoc/doc/Design/umobjects.html#sendingtosources">sending to sources</a>
162   * feature.
163   * See <a href="https://ultramessaging.github.io/lbmct/doc/Domain_ID.html">Domain ID</a> for more information.
164   * <p>
165   * @param domainId  Topic Resolution Domain ID.
166   */
167  @SuppressWarnings("WeakerAccess")  // public API.
168  public void setDomainId(int domainId) { this.domainId = domainId; }
169
170  /**
171   * Override the default value for the configuration object's delay before a receiver sends its the initial connection
172   * request handshake.
173   * <p>
174   * @param delayCreq  Time in milliseconds to delay.
175   */
176  @SuppressWarnings("WeakerAccess")  // public API.
177  public void setDelayCreq(int delayCreq) { this.delayCreq = delayCreq; }
178
179  /**
180   * Override the default value for the configuration object's time between handshake attempts.
181   * When a connected source and receiver are connecting or disconnecting, they exchange handshake messages to
182   * synchronize state.
183   * If a CT source or receiver sends a handshake and expects a response but does not get one within
184   * {@code retryIvl} milliseconds, the operation will be retried.
185   * See also {@link #setMaxTries}.
186   * <p>
187   * @param retryIvl  Time in milliseconds to wait.
188   */
189  @SuppressWarnings("WeakerAccess")  // public API.
190  public void setRetryIvl(int retryIvl) { this.retryIvl = retryIvl; }
191
192  /**
193   * Override the default value for the configuration object's number of tries.
194   * When a connected source and receiver are connecting or disconnecting, they exchange handshake messages to
195   * synchronize state.
196   * If a CT source or receiver sends a handshake and expects a response but does not get one within
197   * {@code retryIvl} milliseconds, the operation will be retried.
198   * After {@code maxTries} attempts, CT will give up, log an error, and stop retrying.
199   * <p>
200   * @param maxTries  Number of times to attempt handshakes.
201   */
202  @SuppressWarnings("WeakerAccess")  // public API.
203  public void setMaxTries(int maxTries) { this.maxTries = maxTries; }
204
205  /**
206   * Override the default value for the configuration object's received message pre-connected delivery behavior.
207   * Normally, a Connected Topics Receiver will not deliver messages received prior to a connection being established.
208   * Setting {@code preDelivery} to 1 enables the CT receiver to deliver messages received outside of the connected
209   * state.
210   * This can enable a CT receiver to get messages from a non-CT source,
211   * although this use case is not recommended. See <a href="https://ultramessaging.github.io/lbmct/#interoperability"></a>.
212   *
213   * @param preDelivery  1=deliver messages outside of connection; 0=do not deliver.
214   */
215  @SuppressWarnings("WeakerAccess")  // public API.
216  public void setPreDelivery(int preDelivery) { this.preDelivery = preDelivery; }
217
218  /**
219   * Gets the current value for the configuration object's test bits.
220   * This is important for cases where the user wants to set or clear an individual bit, leaving the rest of the bits
221   * undisturbed.
222   * @return  Current value for test bits.
223   */
224  @SuppressWarnings("WeakerAccess")  // public API.
225  public int getTestBits() { return testBits; }
226
227  /**
228   * Gets domain ID.
229   * @return  Current domain ID.
230   */
231  @SuppressWarnings("WeakerAccess")  // public API.
232  public int getDomainId() { return domainId; }
233
234  /**
235   * Gets delay before a receiver sends its the initial connection request handshake.
236   * @return Time in milliseconds to delay.
237   */
238  @SuppressWarnings("WeakerAccess")  // public API.
239  public int getDelayCreq() { return delayCreq; }
240
241  /**
242   * Gets time between handshake attempts.
243   * @return  Time in milliseconds between attempts.
244   */
245  @SuppressWarnings("WeakerAccess")  // public API.
246  public int getRetryIvl() { return retryIvl; }
247
248  /**
249   * Gets maximum number of handshake attempts.
250   * @return  Maximum number of attempts.
251   */
252  @SuppressWarnings("WeakerAccess")  // public API.
253  public int getMaxTries() { return maxTries; }
254
255  /**
256   * Gets message delivery with non-connected sources.
257   * @return  1=deliver messages outside of connection; 0=do not deliver.
258   */
259  @SuppressWarnings("WeakerAccess")  // public API.
260  public int getPreDelivery() { return preDelivery; }
