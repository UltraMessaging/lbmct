024import java.nio.*;
025import com.latencybusters.lbm.*;
026
027@SuppressWarnings("WeakerAccess")  // public API.
028public class LbmCtPeerInfo {
029  @SuppressWarnings("WeakerAccess")  // public API.
030  public final static int STATUS_OK=0;
031  @SuppressWarnings("WeakerAccess")  // public API.
032  public final static int STATUS_BAD_CLOSE=-1;
033
034
035  // Flag bits for "flags" field.
036  @SuppressWarnings("WeakerAccess")  // public API.
037  public final static int FLAGS_SRC_METADATA      = 0x01;
038  @SuppressWarnings("WeakerAccess")  // public API.
039  public final static int FLAGS_RCV_METADATA      = 0x02;
040  @SuppressWarnings("WeakerAccess")  // public API.
041  public final static int FLAGS_RCV_SOURCE_STR    = 0x04;
042  @SuppressWarnings("WeakerAccess")  // public API.
043  public final static int FLAGS_RCV_START_SEQ_NUM = 0x08;
044  @SuppressWarnings("WeakerAccess")  // public API.
045  public final static int FLAGS_RCV_END_SEQ_NUM   = 0x10;
046
047  private int status = STATUS_BAD_CLOSE;
048  private int flags = 0;
049  private ByteBuffer srcMetadata = null;
050  private ByteBuffer rcvMetadata = null;
051  private long rcvStartSequenceNumber = 0;
052  private long rcvEndSequenceNumber = 0;
053  private String rcvSourceStr = null;
054
055  // No constructor needed.
056
057  @SuppressWarnings("WeakerAccess")  // public API.
058  public int getStatus() {
059    return status;
060  }
061  @SuppressWarnings("WeakerAccess")  // public API.
062  public int getFlags() { return flags; }
063  @SuppressWarnings("WeakerAccess")  // public API.
064  public ByteBuffer getSrcMetadata() throws Exception {
065    if ((flags & FLAGS_SRC_METADATA) == 0) {
066      throw new LBMException("srcMetadata undefined");
067    }
068    return srcMetadata;
069  }
070  @SuppressWarnings("WeakerAccess")  // public API.
071  public ByteBuffer getRcvMetadata() throws Exception {
072    if ((flags & FLAGS_RCV_METADATA) == 0) {
073      throw new LBMException("rcvMetadata undefined");
074    }
075    return rcvMetadata;
076  }
077  @SuppressWarnings("WeakerAccess")  // public API.
078  public long getRcvStartSequenceNumber() throws Exception {
079    if ((flags & FLAGS_RCV_START_SEQ_NUM) == 0) {
080      throw new LBMException("rcvStartSequenceNumber undefined");
081    }
082    return rcvStartSequenceNumber;
083  }
084  @SuppressWarnings("WeakerAccess")  // public API.
085  public long getRcvEndSequenceNumber() throws Exception {
086    if ((flags & FLAGS_RCV_END_SEQ_NUM) == 0) {
087      throw new LBMException("rcvEndSequenceNumber undefined");
088    }
089    return rcvEndSequenceNumber;
090  }
091  @SuppressWarnings("WeakerAccess")  // public API.
092  public String getRcvSourceStr() throws Exception {
093    if ((flags & FLAGS_RCV_SOURCE_STR) == 0) {
094      throw new LBMException("rcvSourceStr undefined");
095    }
096    return rcvSourceStr;
097  }
098
099  void setStatus(int status) { this.status = status; }
100
101  void setSrcMetadata(ByteBuffer inMetadata) {
102    // Make a deep copy of the metadata.
103    int oldPosition = inMetadata.position();
104    int len = inMetadata.remaining();
105    srcMetadata = ByteBuffer.allocate(len);
106    inMetadata.get(srcMetadata.array(), 0, len);
107    srcMetadata.position(len);
108    srcMetadata.flip();
109
110    inMetadata.position(oldPosition);  // Restore input metadata byte buffer.
111    flags = flags | FLAGS_SRC_METADATA;
112  }
113  void setRcvMetadata(ByteBuffer inMetadata) {
114    // Make a deep copy of the metadata.
115    int oldPosition = inMetadata.position();
116    int len = inMetadata.limit();
117    rcvMetadata = ByteBuffer.allocate(len);
118    inMetadata.get(rcvMetadata.array(), 0, len);
119    rcvMetadata.position(len);
120    rcvMetadata.flip();
121
122    inMetadata.position(oldPosition);  // Restore input metadata byte buffer.
123    flags = flags | FLAGS_RCV_METADATA;
124  }
125  void setRcvStartSequenceNumber(long inStartSequenceNumber) {
126    rcvStartSequenceNumber = inStartSequenceNumber;
127    flags = flags | FLAGS_RCV_START_SEQ_NUM;
128  }
129  void setRcvEndSequenceNumber(long inEndSequenceNumber) {
130    rcvEndSequenceNumber = inEndSequenceNumber;
131    flags = flags | FLAGS_RCV_END_SEQ_NUM;
132  }
133  void setRcvSourceStr(String inSourceStr) {
134    rcvSourceStr = inSourceStr;
135    flags = flags | FLAGS_RCV_SOURCE_STR;
136  }
137}
