024import java.util.*;
025import com.latencybusters.lbm.*;

027/**
028 * A connected receiver.
029 * Maintains an underlying UM receiver.
030 * When an {@code LbmCtRcv} is created, its full initialization is deferred until its {@link #start} method is called.
031 */
032@SuppressWarnings("WeakerAccess")  // public API.
033public class LbmCtRcv {
034  private Object lock = new Object();
035  private LbmCt ct = null;
036  private LbmCtCtrlr ctrlr = null;
037  private LBMContext ctx = null;
038  private LBMReceiver umRcv = null;
039  private String topicStr = null;
040  private LBMReceiverCallback rcvCb = null;
041  private LbmCtRcvConnCreateCallback connCreateCb = null;
042  private LbmCtRcvConnDeleteCallback connDeleteCb = null;
043  private Object cbArg = null;
044  private Set<LbmCtRcvConn> rcvConnSet = null;
045  private int numActiveConns = 0;
046
047  private boolean stopping = false;
048  private LbmCtCtrlrCmd pendingStopCmd = null;
049  private boolean umReceiverStopSubmitted = false;
050  private boolean finalStopSubmitted = false;
051
052  /**
053   * Creates a Connected Topic Receiver object.
054   * Creation of a CT Receiver object creates an underlying UM Receiver.
055   * Unlike a UM receiver, creation of a CT Receiver object is not a two step process.
056   * (A UM receiver requires first looking up a UM topic object, and then creating the receiver.
057   * A CT receiver combines the two steps in its {@link #start} method.)
058   * <p>
059   * This constructor only creates the object.
060   * Its full initialization is deferred until its {@code start} method is called.
061   */
062  public LbmCtRcv() { }
063
064  // Getters.
065  Object getLock() { return lock; }
066  LbmCt getCt() { return ct; }
067  String getTopicStr() { return topicStr; }
068  LBMReceiverCallback getRcvCb() { return rcvCb; }
069  LbmCtRcvConnCreateCallback getConnCreateCb() { return connCreateCb; }
070  LbmCtRcvConnDeleteCallback getConnDeleteCb() { return connDeleteCb; }
071  Object getCbArg() { return cbArg; }
072  boolean isStopping() { return stopping; }
073
074  /**
075   * Initilaizes a CT Receiver object.
076   * This is typically called immediately after the object is created ({@link #LbmCtRcv} constructor).
077   * The {@code start} method can return before any connections are made.
078   * <p>
079   * When the application is finished using this CT Receiver, it should be stopped ({@link #stop} API).
080   * <p>
081   * @param inCt  CT object to associate with this CT Source.
082   * @param topicStr  Topic string.
083   *     This is the same as the {@code symbol} parameter when looking up a UM receiver topic object with
084   *     <a href="https://ultramessaging.github.io/currdoc/doc/JavaAPI/classcom_1_1latencybusters_1_1lbm_1_1LBMTopic.html#aad9c1f2065be4fd8e357f55063268b22">LBMTopic</a>.
085   * @param rcvAttr  Attributes object used to configure the underlying UM receiver.
086   *     This is the same as the {@code lbmrattr} parameter when looking up a UM receiver topic object with
087   *     <a href="https://ultramessaging.github.io/currdoc/doc/JavaAPI/classcom_1_1latencybusters_1_1lbm_1_1LBMTopic.html#aad9c1f2065be4fd8e357f55063268b22">LBMTopic</a>.
088   * @param rcvCb  An object implementing the <a href="https://ultramessaging.github.io/currdoc/doc/JavaAPI/interfacecom_1_1latencybusters_1_1lbm_1_1LBMReceiverCallback.html">LBMReceiverCallback</a>
089   *     interface.
090   *     This is the same as the {@code cb} parameter when creating a UM receiver with
091   *     <a href="https://ultramessaging.github.io/currdoc/doc/JavaAPI/classcom_1_1latencybusters_1_1lbm_1_1LBMReceiver.html#ac913d4f6f0b710711c56e6d1c43596ba">LBMReceiver</a>.
092   *     It is used to deliver messages and other receiver-related events to the application.
093   *     Note that the {@code LBM_BOS} and {@code LBM_EOS} events are filtered out and are not delivered to the
094   *     application.
095   *     The CT connection create and delete callbacks should be used instead.
096   * @param connCreateCb  Callback object invoked when a CT source connects to this CT receiver.
097   *     Technically this parameter is optional (null should be passed if not needed), but its use is central to the
098   *     Connected Topics paradigm.
099   * @param connDeleteCb  Callback object invoked when a CT source disconnects from this CT receiver.
100   *     Technically this parameter is optional (null should be passed if not needed), but its use is central to the
101   *     Connected Topics paradigm.
102   * @param cbArg  Application-specific object to be associated with the connected receiver, which is passed to the
103   *     three callbacks ({@code rcvCb}, {@code connCreateCb}, and {@code connDeleteCb}).
104   *     Note that the CT Receiver maintains a reference to the cbArg object until the CT Receiver is stopped.
105   *     This parameter is optional (null should be passed if not needed).
106   * @throws Exception  LBMException thrown.
107   */
108  public void start(LbmCt inCt, String topicStr, LBMReceiverAttributes rcvAttr, LBMReceiverCallback rcvCb,
109                    LbmCtRcvConnCreateCallback connCreateCb, LbmCtRcvConnDeleteCallback connDeleteCb, Object cbArg)
110      throws Exception
111  {
112    ct = inCt;
113    ctrlr = ct.getCtrlr();
114    ctx = ct.getCtx();
115    LbmCtCtrlrCmd cmd;
116    try {
117      cmd = ctrlr.cmdGet();
118      cmd.setCtRcvStart(this, topicStr, rcvAttr, rcvCb, connCreateCb, connDeleteCb, cbArg);
119      ctrlr.submitWait(cmd);  // This "calls" cmdCtRcvStart below.
120    } catch (Exception e) {
121      throw (new Exception(e));
122    }
123
124    Exception e = cmd.getE();
125    ctrlr.cmdFree(cmd);  // Return command object to free pool.
126    if (e != null) {
127      throw (new Exception(e));
128    }
129  }
130
131  // THREAD: ctrlr
132  boolean cmdCtRcvStart(LbmCtCtrlrCmd cmd) throws Exception {
133    topicStr = cmd.getRcvTopicStr();
134    connCreateCb = cmd.getRcvConnCreateCb();
135    connDeleteCb = cmd.getRcvConnDeleteCb();
136    rcvCb = cmd.getRcvCb();
137    cbArg = cmd.getRcvCbArg();
138    SrcNotification srcNotification = new SrcNotification(this, ct, ctrlr);
139
140    rcvConnSet = new HashSet<>();
141
142    ct.addToCtRcvSet(this);
143
144    LBMReceiverAttributes rcvAttr = cmd.getRcvAttr();
145    if (rcvAttr == null) {
146      // User didn't supply an attribute object, create our own.
147      rcvAttr = new LBMReceiverAttributes();
148    }
149
150    rcvAttr.setSourceNotificationCallbacks(srcNotification, srcNotification, null);
151
152    LBMTopic topicObj = new LBMTopic(ctx, topicStr, rcvAttr);
153    RcvSideMsgRcvCb rcvSideMsgRcvCb = new RcvSideMsgRcvCb(ct, ctrlr);
154    umRcv = new LBMReceiver(ctx, topicObj, rcvSideMsgRcvCb, cbArg);
155
156    if (cmd.getRcvAttr() == null) {
157      // We created our own attribute object; delete it.
158      rcvAttr.dispose();
159    }
160    return true;
161  }
162
163  /**
164   * Stops this CT receiver object.
165   * Calling this method performs a graceful disconnection of any active connections, blocking the caller until
166   * the disconnections are completed.
167   * <p>
168   * @throws Exception  LBMException thrown.
169   */
170  @SuppressWarnings("WeakerAccess")  // public API.
171  public void stop() throws Exception {
172    LbmCtCtrlr ctrlr = this.ctrlr;  // get a local copy since the command will delete the ctSrc.
173
174    LbmCtCtrlrCmd nextCmd = ctrlr.cmdGet();
175    nextCmd.setCtRcvStop(this);
176    ctrlr.submitWait(nextCmd);  // This "calls" cmdCtRcvStop below.
177
178    Exception e = nextCmd.getE();  // Save exception for re-throwing after freeing the command.
179    ctrlr.cmdFree(nextCmd);  // Return command object to free pool.
180    if (e != null) {
181      throw (new Exception(e));
182    }
183
184    // Release any references to other objects.
185    clear();
186  }
187
188  // THREAD: ctrlr
189  boolean cmdCtRcvStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) {
190    synchronized (this) {
191      stopping = true;
192      pendingStopCmd = cmd;
193
194      for (LbmCtRcvConn rcvConn : rcvConnSet) {
195        // These disconnects are submitted to the command queue.
196        rcvConn.disconnect();
197      }
198
199      if ((numActiveConns == 0) && (!umReceiverStopSubmitted)) {
200        // Since stopping is true, no new connections should become active.
201        // There might be some connection-related commands queued.  Queue the UM receiver stop behind them.
202        LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
203        nextCmd.setCtRcvUmReceiverStop(this);
204        ctrlr.submitNowait(nextCmd);
205        umReceiverStopSubmitted = true;
206      }
207    }
208
209    return false;
210  }
211
212  // This is called when there are no more "active" connections.  But there can still be STOP_WAIT connections.
213  // Also, there can be some connection-related commands queued.
214  // THREAD: ctrlr
215  boolean cmdCtRcvUmReceiverStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) throws Exception {
216    // Stop the UM receiver outside of the lock because the call to stop can directly call onSourceDelete(), which
217    // might need to take the lock.
218    if (umRcv != null) {
219      umRcv.close();
220      umRcv = null;
221    }
222
223    // At this point, no new connections will be created.  But there might be some commands in the
224    // queue waiting to be processed.  Submit the final stop to be behind those.
225    synchronized(this) {
226      if (numActiveConns != 0) {
227        // This should never happen.  Log warning and continue.
228        LBMPubLog.pubLog(LBM.LOG_WARNING, "Internal error: cmdCtRcvUmReceiverStop: numActiveConns=" + numActiveConns + "\n");
229      }
230      if (rcvConnSet.isEmpty()) {
231        // No more
232        if (!finalStopSubmitted) {  // Don't submit another one if one is already pending.
233          LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
234          nextCmd.setCtRcvFinalStop(this);
235          ctrlr.submitNowait(nextCmd);  // This "calls" cmdRcvConnStop below.
236
237          finalStopSubmitted = true;
238        }
239      }
240    }
241    return true;
242  }
243
244  // There should be no more connections of any kind, active or STOP_WAIT.
245  // THREAD: ctrlr
246  boolean cmdCtRcvFinalStop(@SuppressWarnings("unused") LbmCtCtrlrCmd cmd) {
247    synchronized (this) {
248      // There should be no connections left at all.
249      try {
250        if (! rcvConnSet.isEmpty()) {
251          throw new LBMException("Internal error, cmdCtRcvFinalStop: rcvConnSet not empty");
252        }
253      } catch (Exception e) {
254        // Pass the exception back to the application.
255        pendingStopCmd.setE(e);
256      }
257
258      ct.removeFromCtRcvSet(this);
259      // This ct receiver is now fully removed from the LbmCt.
260
261      // Wake up ctRcv stop().
262      ctrlr.cmdComplete(pendingStopCmd);
263    }
264    return true;
265  }
266
267  // Release references to other objects.  This is in case the
268  // application is holding a reference to the CT receiver.
269  private void clear() {
270    ct = null;
271    ctrlr = null;
272    ctx = null;
273    umRcv = null;
274    topicStr = null;
275    rcvCb = null;
276    connCreateCb = null;
277    connDeleteCb = null;
278    cbArg = null;
279    rcvConnSet = null;
280    pendingStopCmd = null;
281  }
282
283  void addToRcvConnSet(LbmCtRcvConn rcvConn) {
284    rcvConnSet.add(rcvConn);
285  }
286
287  void removeFromRcvConnSet(LbmCtRcvConn rcvConn) {
288    synchronized (this) {
289      rcvConnSet.remove(rcvConn);
290      if (isStopping() && rcvConnSet.isEmpty()) {
291        if (!umReceiverStopSubmitted) {
292          LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
293          nextCmd.setCtRcvUmReceiverStop(this);
294          ctrlr.submitNowait(nextCmd);
295
296          umReceiverStopSubmitted = true;
297        }
298        else if (!finalStopSubmitted) {  // Don't submit another one if one is already pending.
299          // At this point, no new connections will be created.  But there might be some commands in the
300          // queue waiting to be processed.  Submit the final stop to be behind those.
301          LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
302          nextCmd.setCtRcvFinalStop(this);
303          ctrlr.submitNowait(nextCmd);
304
305          finalStopSubmitted = true;
306        }
307      }
308    }
309  }
310
311  void incrementActiveConns() {
312    synchronized (this) {
313      numActiveConns++;
314    }
315  }
316
317  void decrementActiveConns() {
318    synchronized (this) {
319      numActiveConns--;
320      if (isStopping()) {
321        if ((numActiveConns == 0) && (!umReceiverStopSubmitted)) {
322          // There might be some connection-related commands queued.  Queue the UM receiver stop behind.
323          LbmCtCtrlrCmd nextCmd = ct.getCtrlr().cmdGet();
324          nextCmd.setCtRcvUmReceiverStop(this);
325          ctrlr.submitNowait(nextCmd);
326          umReceiverStopSubmitted = true;
327        }
328      }
329    }
330  }

332  private static class SrcNotification implements LBMSourceCreationCallback, LBMSourceDeletionCallback {
333    LbmCtRcv ctRcv;
334    LbmCt ct;
335    LbmCtCtrlr ctrlr;
336
337    private SrcNotification(LbmCtRcv ctRcv, LbmCt ct, LbmCtCtrlr ctrlr) {
338      this.ctRcv = ctRcv;
339      this.ct = ct;
340      this.ctrlr = ctrlr;
341    }
342
343    // Not part of public API.  Declared public to conform to interface.
344    // THREAD: ctx
345    public Object onNewSource(String sourceStr, Object cbObj) {
346      LbmCtRcvConn rcvConn = new LbmCtRcvConn(ctRcv);
347
348      try {
349        ct.dbg("onNewSource: " + rcvConn);
350        rcvConn.start(sourceStr);
351      } catch (Exception e) {
352        LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCtRcv:OnNewSource(" + sourceStr + "): " + LbmCt.exDetails(e) + "\n");
353      }
354      // Return the rcvConn object as the per-source callback argument.
355      return rcvConn;
356    }
357
358    // Not part of public API.  Declared public to conform to interface.
359    // THREAD: ctx
360    public int onSourceDelete(String source, Object cbObj, Object sourceCbObj) {
361      LbmCtRcvConn rcvConn = (LbmCtRcvConn)sourceCbObj;
362
363      ct.dbg("onSourceDelete: " + rcvConn);
364      try {
365        rcvConn.stop();
366      } catch (Exception e) {
367        LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCtRcv:onSourceDelete: " + LbmCt.exDetails(e) + "\n");
368      }
369      return 0;
370    }
371  }

373  private static class RcvSideMsgRcvCb implements LBMReceiverCallback {
374    LbmCt ct;
375    LbmCtCtrlr ctrlr;
376
377    private RcvSideMsgRcvCb(LbmCt ct, LbmCtCtrlr ctrlr) {
378      this.ct = ct;
379      this.ctrlr = ctrlr;
380    }
381
382    // Not part of public API.  Declared public to conform to interface.
383    // THREAD: ctx
384    public int onReceive(Object cbArgs, LBMMessage umMsg) {
385      // Connected Topics receivers don't use BOS/EOS for anything.
386      if ((umMsg.type() == LBM.MSG_BOS) || (umMsg.type() == LBM.MSG_EOS)) {
387        umMsg.dispose();  // Ignore.
388      } else if (umMsg.sourceClientObject() == null) {
389        LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCtRcv.onReceive: topic=" + umMsg.topicName() + ", sourceClientObject is null\n");
390        umMsg.dispose();
391      } else {
392        try {
393          LbmCtRcvConn rcvConn = (LbmCtRcvConn)umMsg.sourceClientObject();
394          ct.dbg("onReceive: " + rcvConn);
395          // Have rcvConn.  Continue handling inside the conn object.
396          rcvConn.handleMsg(umMsg);
397          // Don't dispose message here because most received messages are for user application, and the app gets to
398          // decide the message disposition.
399        } catch (Exception e) {
400          LBMPubLog.pubLog(LBM.LOG_WARNING, "LbmCtRcv.onReceive: " + LbmCt.exDetails(e) + "\n");
401        }
402      }
403      return 0;
404    }
405  }
